AngularJS - Live search

March 12, 2018

I love to see web pages & apps with the live search feature. In fact, they are perfect for SPAs (Single Page Applications). My MovieFlix app written in MEAN has something similar. I am gonna share something very close to that here. The only difference you’ll notice is the use of scope in angular & where it applies i.e. change & filter. Also, this is not a tutorial post, so don’t expect any explanations here. However, the markup & the javascript is really easy & straightforward.

index.html

<!doctype html>
<html ng-app="liveSearch">
<head>
  <meta charset="utf-8">
  <title>AngularJS Live Search</title>
</head>

<body ng-controller="searchController">
  <h1>Angular Live Search</h1>

  <input type="text" placeholder="Search" ng-model="query" ng-focus="focus=true">
  <div id="search-results" ng-show="focus">
    <div class="search-result" ng-repeat="item in data | search:query" ng-bind="item" ng-click="setQuery(item)"></div>
  </div>

  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.7.2/angular.min.js"></script>
  <script src="data.js"></script>
  <script src="app.js"></script>
</body>
</html>

AngularJS code:

var app = angular.module('liveSearch', []);

// Search controller
app.controller('searchController', function($scope) {
  $scope.data = data;

  $scope.setQuery = function(query) {
    $scope.query = query;
    $scope.focus = false;
  };
});

// Filtering the data
app.filter('search', function() {
  return search;
});

// Matching query & items from our array of items in data.js; convert to lower case while doing so.
function search(arr, query) {
  if (!query) {
    return arr;
  }

  var res = [];
  query = query.toLowerCase();

  angular.forEach(arr, function(item) {
    if (item.toLowerCase().indexOf(query) !== -1) {
      res.push(item);
    }
  });

  return res;
};
